<!--
    - Policies are applied in the order they appear.
    - Position <base/> inside a section to inherit policies from the outer scope.
    - Comments within policies are not preserved.
-->
<!-- Add policies as children to the <inbound>, <outbound>, <backend>, and <on-error> elements -->
<policies>
    <!-- Throttle, authorize, validate, cache, or transform the requests -->
    <inbound>
        <!-- Always place the cors policy first -->
        <cors allow-credentials="true">
            <allowed-origins>
                <origin>https://travel.allwyns-lab.xyz</origin>
            </allowed-origins>
            <allowed-methods preflight-result-max-age="300">
                <method>GET</method>
                <method>POST</method>
                <method>PUT</method>
                <method>DELETE</method>
                <method>OPTIONS</method>
            </allowed-methods>
            <allowed-headers>
                <header>*</header>
            </allowed-headers>
        </cors>
        <base />
        <rewrite-uri template="/planTrip" />
        <!-- 2. Inject the secret function key natively, instead of 'code' parameter -->
        <set-header name="x-functions-key" exists-action="override">
            <value>{{travelagentapp-key}}</value>
        </set-header>
        <set-header name="x-apim-gateway-secret" exists-action="override">
            <value>{{mygateway-secret-name}}</value>
        </set-header>
        <validate-jwt header-name="Authorization" failed-validation-httpcode="401" failed-validation-error-message="Your token is missing or not valid" require-expiration-time="true" require-scheme="Bearer" output-token-variable-name="jwt">
            <openid-config url="https://securetoken.google.com/{{firebaseapp-travelexp-project-id}}/.well-known/openid-configuration" />
            <audiences>
                <audience>{{firebaseapp-travelexp-project-id}}</audience>
            </audiences>
            <issuers>
                <issuer>https://securetoken.google.com/{{firebaseapp-travelexp-project-id}}</issuer>
            </issuers>
        </validate-jwt>
        <!-- Forward claims to Azure Function backend -->
        <set-header name="X-User-Id" exists-action="override">
            <value>@(context.Variables.GetValueOrDefault<Jwt>("jwt").Claims.GetValueOrDefault("sub", ""))</value>
        </set-header>
        <set-header name="X-User-Email" exists-action="override">
            <value>@(context.Variables.GetValueOrDefault<Jwt>("jwt").Claims.GetValueOrDefault("email", ""))</value>
        </set-header>
        <set-header name="X-User-Name" exists-action="override">
            <value>@(context.Variables.GetValueOrDefault<Jwt>("jwt").Claims.GetValueOrDefault("name", ""))</value>
        </set-header>
    </inbound>
    <!-- Control if and how the requests are forwarded to services  -->
    <backend>
        <base />
    </backend>
    <!-- Customize the responses -->
    <outbound>
        <base />
    </outbound>
    <!-- Handle exceptions and customize error responses  -->
    <on-error>
        <base />
    </on-error>
</policies>
